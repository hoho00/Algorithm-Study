#include<bits/stdc++.h>
using namespace std;
int main(){
	ios_base::sync_with_stdio(false);
	//freopen("input.txt", "rt", stdin);
	int n;
	cin>>n;
	vector<int> cost(n), dy(n);
	for(int i=0; i<n; i++){
		cin>>cost[i];
	}
	dy[0]=cost[0];
	dy[1]=cost[1];
	for(int i=2; i<n; i++){
		dy[i]=min(dy[i-1], dy[i-2])+cost[i];
	}
	int answer=min(dy[n-1], dy[n-2]);
	cout<<answer;
	return 0;
}
